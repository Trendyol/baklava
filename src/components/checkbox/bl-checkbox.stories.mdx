import { html } from 'lit';
import { ifDefined } from 'lit/directives/if-defined.js';
import { Meta, Canvas, ArgsTable, Story } from '@storybook/addon-docs';

<Meta
  title="Components/Checkbox"
  component="bl-checkbox"
  argTypes={{
    label: {
      control: ''
    },
    disabled: {
      control: false
    },
    checked: {
      control: false
    },
    indeterminate: {
      control: false
    }
  }}
/>

export const CheckboxTemplate = (args) => html`
  <bl-checkbox
    disabled="${ifDefined(args.disabled)}"
    checked="${ifDefined(args.checked)}"
    indeterminate="${ifDefined(args.indeterminate)}">
    ${args.label}
  </bl-checkbox>
`;

export const DisabledChecboxesTemplate = (args) => html`
${CheckboxTemplate({label: 'checked', disabled: true, checked: true, ...args})}
${CheckboxTemplate({label: 'unchecked', disabled: true, ...args})}`;

# Checkbox
Checkbox component can be used to control checked / unchecked statuses.

### Usage
* A checkbox is not required to have a label
* Label is right aligned
* Single checkbox can be indeterminate
* A checkbox that has both checked and indeterminate state will have only checked state

## Basic
You can show label by just using slot.
<Canvas>
  <Story name="Basic Usage" args={{ label: 'checkbox' }}>
    {CheckboxTemplate.bind({})}
  </Story>
</Canvas>

## Checked
You can set checked state by just using checked attribute.
<Canvas>
  <Story name="Checked" args={{ label: 'checkbox', checked: true }}>

</Canvas>

## Indeterminate
You can set indeterminate state by just using indeterminate attribute.
<Canvas>
  <Story name="Indeterminate" args={{ label: 'checkbox', indeterminate: true }}>
    {CheckboxTemplate.bind({})}
  </Story>
</Canvas>

## Disabled
You can set disabled state by just using disabled attribute.

  <Story name="Disabled">
